import {Meta, Stylesheet} from "@canopy-iiif/app/head";
import {CanopyHeader, CanopyFooter} from "@canopy-iiif/app/ui";

export function Head() {
  return (
    <>
      <Meta />
      <Stylesheet />
      <script dangerouslySetInnerHTML={{__html: `
(function(){
  var PREFIX = /^Environmental Impact Statement,\\s+/i;
  var SUFFIX = /\\s*:\\s*environmental impact statement\\.?\\s*$/i;
  function clean(t){ return t.replace(PREFIX,'').replace(SUFFIX,''); }
  function fix(el){
    el.childNodes.forEach(function(n){
      if(n.nodeType===3){
        var c=clean(n.textContent);
        if(c!==n.textContent) n.textContent=c;
      } else if(n.nodeType===1){ fix(n); }
    });
  }
  document.addEventListener('DOMContentLoaded',function(){
    fix(document.body);
    document.title=clean(document.title);
  });
  new MutationObserver(function(ms){
    ms.forEach(function(m){
      m.addedNodes.forEach(function(n){ if(n.nodeType===1) fix(n); });
    });
  }).observe(document.documentElement,{childList:true,subtree:true});
})();
`}} />
      <link rel="icon" href="/favicon.png" type="image/png" />
      <link rel="preconnect" href="https://fonts.googleapis.com" />
      <link
        rel="preconnect"
        href="https://fonts.gstatic.com"
        crossOrigin="true"
      />
      <link
        href="https://fonts.googleapis.com/css2?family=Fraunces:ital,opsz,wght,SOFT,WONK@0,9..144,300..600,50,0..1;1,9..144,300..600,50,0..1&display=swap"
        rel="stylesheet"
      />
    </>
  );
}
